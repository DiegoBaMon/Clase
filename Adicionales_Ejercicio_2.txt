import random

# Inicializamos el contador de veces que sale el número 1
contador_uno = 0
total_lanzamientos = 50

print("Iniciando los 50 lanzamientos del dado...\n")

for lanzamiento in range(1, total_lanzamientos + 1):
    # Genera un número aleatorio entre 1 y 6
    resultado = random.randint(1, 6)
    print(f"Lanzamiento {lanzamiento:02d}: Salió el {resultado}")
    
    # Si el resultado es 1, sumamos al contador
    if resultado == 1:
        contador_uno += 1

print("-" * 30)
print(f"El número 1 salió un total de: {contador_uno} veces.")
