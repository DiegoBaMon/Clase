# Pedimos el tamaño (la mitad del diamante, ej: 5)
n = int(input("Introduce el tamaño del diamante (mitad superior): "))

print("\nResultado:")

# 1. Mitad superior (incluyendo la línea del centro)
for i in range(1, n + 1):
    espacios = " " * (n - i)
    asteriscos = "*" * (2 * i - 1)
    print(espacios + asteriscos)

# 2. Mitad inferior (va en orden descendente)
for i in range(n - 1, 0, -1):
    espacios = " " * (n - i)
    asteriscos = "*" * (2 * i - 1)
    print(espacios + asteriscos)
